from parceldata import ParcelData

pd = ParcelData(api_key="pk_live_...")

# Lookup by address
prop = pd.lookup("123 Main St, Austin, TX")
print(prop.zoning)       # "R1 - Single Family"
print(prop.lot_sqft)     # 8500
print(prop.tax_assessed) # 425000

# Search with filters
results = pd.search(
    city="Austin",
    min_value=300000,
    zoning="R1",
    limit=25
)
# Lookup by address
curl https://api.parceldata.ai/v1/properties/address \
  -H "Authorization: Bearer pk_live_..." \
  -d "street=123 Main St" \
  -d "city=Austin" \
  -d "state=TX"

# Search properties
curl -X POST https://api.parceldata.ai/v1/properties/search \
  -H "Authorization: Bearer pk_live_..." \
  -H "Content-Type: application/json" \
  -d '{
    "city": "Austin",
    "min_value": 300000,
    "zoning": "R1",
    "limit": 25
  }'
